# To create react app run this command
npx create-react-app my_reads_app


# To start the application run this command
npm start

# To install react routes package run this command 
npm i react-router-dom


# To install react icons run this command
npm i react-icons --save

# To install Bootstrap and react Bootstrap run this command 
npm i react-bootstrap bootstrap

# To add alert 
npm i --save sweetalert2 sweetalert2-react-content
